%   ── Chapter Head — navy banner matching title-page colour tone ─────────────
%   (Bypasses titlesec for \chapter to avoid the "Nested titles" error;
%    titlesec is still used for \section and \subsection below.)
\makeatletter
\renewcommand{\@makechapterhead}[1]{%
  {\parindent\z@
   \setlength{\fboxsep}{1.2em}%
   \noindent\colorbox{DeepSpace}{%
     \begin{minipage}{\dimexpr\textwidth-2.4em\relax}%
       \vspace{0.3em}%
       {\small\color{StarGold!80}\scshape chapter\ \thechapter}%
       \par\vspace{0.3em}%
       {\Large\bfseries\color{white}#1}%
       \par\vspace{0.35em}%
       {\color{StarGold!35}\rule{\linewidth}{0.4pt}}%
       \par\vspace{0.3em}%
     \end{minipage}%
   }\par\vspace{2em}%
  }%
}
\renewcommand{\@makeschapterhead}[1]{%
  {\parindent\z@
   \setlength{\fboxsep}{1.2em}%
   \noindent\colorbox{DeepSpace}{%
     \begin{minipage}{\dimexpr\textwidth-2.4em\relax}%
       \vspace{0.3em}%
       {\Large\bfseries\color{white}#1}%
       \par\vspace{0.4em}%
     \end{minipage}%
   }\par\vspace{2em}%
  }%
}
\makeatother
